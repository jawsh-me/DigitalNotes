%1	TTY

%M	Full Hostname
%m	Hostname
%n	Username
%#	Shell Marker (#/%)
%~	Dir
&T	24hr Time

%f	End FG Color
%K	BG Color
%k	End BG Color

%W	mm/dd/yy
%w	DoW dd (Mon 18)

	%A	Full Day of Week
	$B	Full Month
	%d	Day of month
	%H	24hr hour
	%M	Minute
	%R	24hr hour+minute
